Type System #2

public function addPerson(options : {
  name : String,
  company : String,
  ?age : Int
}) { /* do something here */}


company.addPerson({
  name : "Franco", // string/not optional
  company : "Pellucid", // same
  age : null, // integer/optional
});

String Interpolation

return
  'My name is ' + name.toUpperCase() +
  ' and I am ' + age + ' old';


return
  'My name is ${name.toUpperCase()} and I am $age old';

Array Comprehension

var arr = [];
for(var i = 0; i < 5; i++) {
  arr.push(i*2);
}


var arr = [for(i in 0...5) i*2];
